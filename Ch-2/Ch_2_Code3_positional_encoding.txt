import numpy as np

def getPositionEncoding(seq_len, d, n=10000):
    """
    Generate sinusoidal positional encoding matrix.
    
    Args:
        seq_len (int): Length of the sequence (number of positions).
        d (int): Embedding dimension size.
        n (int): Scaling factor for frequency (default 10000).
    
    Returns:
        np.ndarray: Positional encoding matrix of shape (seq_len, d)
    """
    P = np.zeros((seq_len, d))
    for pos in range(seq_len):
        for i in range(d // 2):
            denominator = np.power(n, 2 * i / d)
            P[pos, 2 * i] = np.sin(pos / denominator)
            P[pos, 2 * i + 1] = np.cos(pos / denominator)
    return P
